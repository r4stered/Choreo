132 frc2::CommandPtr Cmd() {
133 if (trajectory.samples.size() == 0) {
134 return frc2::cmd::RunOnce([this] {
135 FRC_ReportError(frc::warn::Warning,

231 frc2::Trigger AtTime(units::second_t timeSinceStart) {
232 if (timeSinceStart < 0_s) {
233 FRC_ReportError(frc::warn::Warning,
234 "Trigger time cannot be negative for {}", name);
235 return offTrigger;
236 }
237
238 if (timeSinceStart > TotalTime()) {
239 FRC_ReportError(
240 frc::warn::Warning,
241 "Trigger time cannout be greater than total trajectory time for {}",
242 name);
243 return offTrigger;
244 }
245
246 return frc2::Trigger(loop,
247 [this, timeSinceStart, triggered = false]() mutable {
248 if (!isActive) {
249 return false;
250 }
251 if (triggered) {
252 return false;
253 }
254 if (TimeIntoTraj() >= timeSinceStart) {
255 triggered = true;
256 return true;
257 }
258 return false;
259 });
260 }

273 frc2::Trigger AtTime(std::string_view eventName) {
274 bool foundEvent = false;
275 frc2::Trigger trig = offTrigger;
276
277 for (const auto& event : trajectory.GetEvents(eventName)) {
278 trig = frc2::Trigger{trig || AtTime(event.timestamp)};
279 foundEvent = true;
280 }
281
282 if (!foundEvent) {
283 FRC_ReportError(frc::warn::Warning, "Event \"{}\" not found for {}",
284 eventName, name);
285 }
286
287 return trig;
288 }

302 frc2::Trigger AtPose(std::string_view eventName,
303 units::meter_t tolerance = DEFAULT_TOLERANCE) {
304 bool foundEvent = false;
305 frc2::Trigger trig = offTrigger;
306
307 for (const auto& event : trajectory.GetEvents(eventName)) {
308 frc::Pose2d pose =
309 trajectory
310 .template SampleAt<Year>(event.timestamp, mirrorTrajectory())
311 .GetPose();
312 trig = frc2::Trigger(trig || AtPose(pose, tolerance));
313 foundEvent = true;
314 }
315
316 if (!foundEvent) {
317 FRC_ReportError(frc::warn::Warning, "Event \"{}\" not found for {}",
318 eventName, name);
319 }
320
321 return trig;
322 }

356 void CmdInitialize() {
357 timer.Restart();
358 isActive = true;
359 timeOffset = 0.0_s;
360 for (auto& event : scheduledEvents) {
361 event.hasTriggered = false;
362 }
363 LogTrajectory(true);

366 void CmdExecute() {
367 auto sampleOpt =
368 trajectory.template SampleAt<Year>(TimeIntoTraj(), mirrorTrajectory());
369 controller(poseSupplier(), sampleOpt.value());
370 currentSample = sampleOpt.value();
371 }
372
373 void CmdEnd(bool interrupted) {
374 timer.Stop();
375 if (interrupted) {
376 controller(currentSample.GetPose(), currentSample);
377 } else {
378 controller(poseSupplier(), trajectory.GetFinalSample().value());
379 }
380 isActive = false;
381 LogTrajectory(false);
382 }
383
384 bool CmdIsFinished() { return TimeIntoTraj() > TotalTime(); }

386 frc2::Trigger AtPose(frc::Pose2d pose, units::meter_t tolerance) {
387 frc::Translation2d checkedTrans =
388 mirrorTrajectory()
389 ? frc::Translation2d{util::fieldLength - pose.Translation().X(),
390 pose.Translation().Y()}
391 : pose.Translation();
392 return frc2::Trigger{
393 loop, [this, checkedTrans, tolerance] {
394 frc::Translation2d currentTrans = poseSupplier().Translation();
395 return currentTrans.Distance(checkedTrans) < tolerance;
396 }};
397 }

399 void AddScheduledEvent(std::string_view eventName,
400 std::function<frc2::CommandPtr()> cmdFactory) {
401 for (const auto& event : trajectory.GetEvents(eventName)) {
402 scheduledEvents.push_back({event.timestamp, std::string(eventName),
403 cmdFactory, frc2::cmd::None(), false});
404 }
405 }
406
407 void CheckAndTriggerEvents() {
408 auto currentTime = TimeIntoTraj();
409 for (auto& event : scheduledEvents) {
410 if (!event.hasTriggered && isActive && currentTime >= event.triggerTime) {
411 event.hasTriggered = true;
412 event.command = event.commandFactory();
413 event.command.Schedule();
414 }
415 }
416 }
