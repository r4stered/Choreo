34 static choreo::ProjectFile GetProjectFile() {
35 if (LAZY_PROJECT_FILE.has_value()) {
36 return LAZY_PROJECT_FILE.value();
37 }
38
39 std::vector<std::filesystem::path> matchingFiles;
40
41 try {
42 for (const auto& entry :
43 std::filesystem::directory_iterator(CHOREO_DIR)) {
44 if (std::filesystem::is_regular_file(entry) &&
45 entry.path().extension() == TRAJECTORY_FILE_EXTENSION) {
46 matchingFiles.push_back(entry.path());
47 }
48 }
49
50 if (matchingFiles.size() == 0) {
51 FRC_ReportError(frc::warn::Warning,
52 "Could not find file in deploy directory!");
53 } else if (matchingFiles.size() > 1) {
54 FRC_ReportError(frc::warn::Warning,
55 "Found multiple project files in deploy directory!");
56 }
57
58 auto fileBuffer = wpi::MemoryBuffer::GetFile(matchingFiles[0].string());
59 if (!fileBuffer) {
60 FRC_ReportError(frc::warn::Warning,
61 "Could not open choreo project file");
62 }
63
64 wpi::json json = wpi::json::parse(fileBuffer.value()->GetCharBuffer());
65 std::string version = json["version"];
66 if (kSpecVersion != version) {
67 throw fmt::format(".chor project file: Wrong version {}. Expected {}",
68 version, kSpecVersion);
69 }
70 choreo::ProjectFile resultProjectFile;
71 choreo::from_json(json, resultProjectFile);
72 LAZY_PROJECT_FILE = resultProjectFile;
73 } catch (const std::filesystem::filesystem_error&) {
74 FRC_ReportError(frc::warn::Warning, "Error finding choreo directory!");
75 } catch (const wpi::json::exception&) {
76 FRC_ReportError(frc::warn::Warning, "Error parsing choreo project file!");
77 }
78 return LAZY_PROJECT_FILE.value();
79 }

91 template <choreo::TrajectorySample SampleType>
92 static std::optional<choreo::Trajectory<SampleType>> LoadTrajectory(
93 std::string_view trajectoryName) {
94 if (trajectoryName.ends_with(TRAJECTORY_FILE_EXTENSION)) {
95 trajectoryName = trajectoryName.substr(
96 0, trajectoryName.size() - TRAJECTORY_FILE_EXTENSION.size());
97 }
98
99 std::string trajectoryFileName = fmt::format(
100 "{}/{}{}", CHOREO_DIR, trajectoryName, TRAJECTORY_FILE_EXTENSION);
101
102 auto fileBuffer = wpi::MemoryBuffer::GetFile(trajectoryFileName);
103 if (!fileBuffer) {
104 FRC_ReportError(frc::warn::Warning, "Could not find trajectory file: {}",
105 trajectoryName);
106 return {};
107 }
108
109 try {
110 return LoadTrajectoryString<SampleType>(
111 std::string{fileBuffer.value()->GetCharBuffer().data(),
112 fileBuffer.value()->size()},
113 trajectoryName);
114 } catch (wpi::json::parse_error& ex) {
115 FRC_ReportError(frc::warn::Warning, "Could not parse trajectory file: {}",
116 trajectoryName);
117 FRC_ReportError(frc::warn::Warning, "{}", ex.what());
118 return {};
119 }
120 return {};
121 }

133 template <choreo::TrajectorySample SampleType>
134 static std::optional<choreo::Trajectory<SampleType>> LoadTrajectoryString(
135 std::string_view trajectoryJsonString, std::string_view trajectoryName) {
136 wpi::json json = wpi::json::parse(trajectoryJsonString);
137 std::string version = json["version"];
138 if (kSpecVersion != version) {
139 throw fmt::format("{}.traj: Wrong version {}. Expected {}",
140 trajectoryName, version, kSpecVersion);
141 }
142 choreo::Trajectory<SampleType> trajectory;
143 choreo::from_json(json, trajectory);
144 return trajectory;
145 }
