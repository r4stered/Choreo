157 bool operator==(const SwerveSample& other) const {
158 constexpr double epsilon = 1e-6;
159
160 auto compare_units = [epsilon](const auto& a, const auto& b) {
161 using UnitType =
162 std::remove_const_t<std::remove_reference_t<decltype(a)>>;
163 return units::math::abs(a - b) < UnitType(epsilon);
164 };
165
166 auto compare_arrays = [&compare_units](const auto& arr1, const auto& arr2) {
167 return std::equal(arr1.begin(), arr1.end(), arr2.begin(), compare_units);
168 };
169
170 return compare_units(timestamp, other.timestamp) &&
171 compare_units(x, other.x) && compare_units(y, other.y) &&
172 compare_units(heading, other.heading) &&
173 compare_units(vx, other.vx) && compare_units(vy, other.vy) &&
174 compare_units(omega, other.omega) && compare_units(ax, other.ax) &&
175 compare_units(ay, other.ay) && compare_units(alpha, other.alpha) &&
176 compare_arrays(moduleForcesX, other.moduleForcesX) &&
177 compare_arrays(moduleForcesY, other.moduleForcesY);
178 }
