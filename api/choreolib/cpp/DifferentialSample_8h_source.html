127 bool operator==(const DifferentialSample& other) const {
128 constexpr double epsilon = 1e-6;
129
130 auto compare_units = [epsilon](const auto& a, const auto& b) {
131 using UnitType =
132 std::remove_const_t<std::remove_reference_t<decltype(a)>>;
133 return units::math::abs(a - b) < UnitType(epsilon);
134 };
135
136 return compare_units(timestamp, other.timestamp) &&
137 compare_units(x, other.x) && compare_units(y, other.y) &&
138 compare_units(heading, other.heading) &&
139 compare_units(vl, other.vl) && compare_units(vr, other.vr) &&
140 compare_units(al, other.al) && compare_units(ar, other.ar) &&
141 compare_units(fl, other.fl) && compare_units(fr, other.fr);
142 }
