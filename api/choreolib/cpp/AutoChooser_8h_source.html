48template <choreo::TrajectorySample SampleType, int Year = util::kDefaultYear>
49class AutoChooser {
50 public:
59 AutoChooser(AutoFactory<SampleType, Year>& factory,
60 std::string_view tableName)
61 : factory{factory} {
62 std::string path =
63 nt::NetworkTable::NormalizeKey(tableName, true) + "/AutoChooser";
64 std::shared_ptr<nt::NetworkTable> table =
65 nt::NetworkTableInstance::GetDefault().GetTable(path);
66
67 selected = table->GetStringTopic("selected").GetEntry(NONE_NAME);
68
69 metaType = table->GetStringTopic(".type").Publish();
70 metaType.Set("String Chooser");
71
72 defaultChoice = table->GetStringTopic("default").Publish();
73 defaultChoice.Set(NONE_NAME);
74
75 active = table->GetStringTopic("active").GetEntry(NONE_NAME);
76
77 std::vector<std::string> keys;
78
79 AddAutoRoutine(NONE_NAME, [] { return frc2::cmd::None(); });
80
81 for (const auto& pair : autoRoutines) {
82 keys.push_back(pair.first);
83 }
84
85 options = table->GetStringArrayTopic("options").GetEntry(keys);
86 }
87
95 void Update() {
96 if (frc::DriverStation::IsDisabled() || frc::RobotBase::IsSimulation()) {
97 std::string selectedString = selected.Get();
98 if (selectedString == lastAutoRoutineName) {
99 return;
100 }
101 if (!autoRoutines.contains(selectedString)) {
102 selected.Set(NONE_NAME);
103 selectedString = NONE_NAME;
104 FRC_ReportError(frc::warn::Warning,
105 "Selected an auto that isn't an option!");
106 }
107 lastAutoRoutineName = selectedString;
108 active.Set(lastAutoRoutineName);
109 }
110 }
111
127 void AddAutoRoutine(std::string name,
128 std::function<frc2::CommandPtr()> generator) {
129 autoRoutines[name] = generator;
130
131 std::vector<std::string> keys;
132 for (const auto& pair : autoRoutines) {
133 keys.push_back(pair.first);
134 }
135
136 options.Set(keys);
137 }
138
144 void Choose(std::string_view choice) {
145 selected.Set(choice);
146 Update();
147 }
148
154 frc2::CommandPtr GetSelectedAutoRoutine() {
155 return autoRoutines[lastAutoRoutineName]();
156 }
157
158 private:
159 static inline std::string NONE_NAME = "Nothing";
160 std::unordered_map<std::string, std::function<frc2::CommandPtr()>>
161 autoRoutines;
162
163 nt::StringEntry selected;
164 nt::StringEntry active;
165
166 nt::StringPublisher metaType;
167 nt::StringPublisher defaultChoice;
168
169 nt::StringArrayEntry options;
170
171 AutoFactory<SampleType, Year>& factory;
172
173 std::string lastAutoRoutineName = NONE_NAME;
174 frc2::CommandPtr lastAutoRoutine{frc2::cmd::None()};
175};
176} // namespace choreo
