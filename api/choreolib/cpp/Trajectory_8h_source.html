214 std::optional<Trajectory<SampleType>> GetSplit(int splitIndex) const {
215 if (splitIndex < 0 || splitIndex >= splits.size()) {
216 return std::nullopt;
217 }
218
219 int start = splits[splitIndex];
220 int end = (splitIndex + 1 < splits.size()) ? splits[splitIndex + 1] + 1
221 : samples.size();
222
223 auto sublist =
224 std::vector<SampleType>(samples.begin() + start, samples.begin() + end);
225 units::second_t startTime = sublist.front().GetTimestamp();
226 units::second_t endTime = sublist.back().GetTimestamp();
227
228 auto offsetSamples =
229 sublist | std::views::transform([startTime](const SampleType& s) {
230 return s.OffsetBy(-startTime);
231 });
232
233 auto filteredEvents =
234 events | std::views::filter([startTime, endTime](const auto& e) {
235 return e.timestamp >= startTime && e.timestamp <= endTime;
236 }) |
237 std::views::transform(
238 [startTime](const auto& e) { return e.OffsetBy(-startTime); });
239
240 return Trajectory<SampleType>{
241 name + "[" + std::to_string(splitIndex) + "]",
242 std::vector<SampleType>(offsetSamples.begin(), offsetSamples.end()),
243 {},
244 std::vector<EventMarker>(filteredEvents.begin(), filteredEvents.end())};
245 }

292 std::optional<SampleType> SampleInternal(units::second_t timestamp) const {
293 if (timestamp < samples[0].GetTimestamp()) {
294 return GetInitialState();
295 }
296 if (timestamp >= GetTotalTime()) {
297 return GetFinalSample();
298 }
299
300 int low = 0;
301 int high = samples.size() - 1;
302
303 while (low != high) {
304 int mid = (low + high) / 2;
305 if (samples[mid].GetTimestamp() < timestamp) {
306 low = mid + 1;
307 } else {
308 high = mid;
309 }
310 }
311
312 if (low == 0) {
313 return samples[low];
314 }
315
316 SampleType behindState = samples[low - 1];
317 SampleType aheadState = samples[low];
318
319 if ((aheadState.GetTimestamp() - behindState.GetTimestamp()) < 1e-6_s) {
320 return aheadState;
321 }
322
323 return behindState.Interpolate(aheadState, timestamp);
324 }
