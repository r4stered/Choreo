161 void SgmtConstraint(size_t fromIndex, size_t toIndex,
162 const Constraint& constraint) {
163 assert(fromIndex < toIndex);
164
165 NewWpts(toIndex);
166 path.waypoints.at(fromIndex).waypointConstraints.push_back(constraint);
167 for (size_t index = fromIndex + 1; index <= toIndex; ++index) {
168 path.waypoints.at(index).waypointConstraints.push_back(constraint);
169 path.waypoints.at(index).segmentConstraints.push_back(constraint);
170 }
171 }

237 void NewWpts(size_t finalIndex) {
238 if (finalIndex < path.waypoints.size()) {
239 return;
240 }
241
242 for (size_t i = path.waypoints.size(); i <= finalIndex; ++i) {
243 path.waypoints.emplace_back();
244 initialGuessPoints.emplace_back(std::vector<Pose2d>{{0.0, 0.0, {0.0}}});
245 if (i != 0) {
246 controlIntervalCounts.push_back(40);
247 }
248 }
249 }
